213  public static bool is_ok(eUserRight right)
214  {
215  int user_id = g.login_user_id;
216  int site_id = g.selected_site_id;
217  if (site_id == 0)
218  return false;
219  var u = g.user_list.Find(p => p.user_id == user_id);
220  if (u == null)
221  return false;
222  // 수퍼 관리자 권한을 가지고 있으면 무조건 Pass
223  if (u.user_group == "S")
224  return true;
225  // 사이트별 사용자리스트에 없으면 False
226  var su = g.site_user_list.Find(p => (p.site_id == site_id) && (p.user_id == user_id));
227  if (su == null)
228  return false;
229  if (su.user_right == "A")
230  {
231  var list = _admin_func_list.Where(p => p == right);
232  return list.Count() != 0;
233  }
234  else
235  {
236  var list = _user_func_list.Where(p => p == right);
237  return list.Count() != 0;
238  }
