71  private static double CalculateNewScrollOffset(
72  double dblViewport_Left,
73  double dblViewport_Right,
74  double dblChild_Left,
75  double dblChild_Right
76  )
77  {
78  //Retrieve basic information about the position of the Viewport within the Extent of the control.
79  bool fIsFurtherToLeft = (dblChild_Left < dblViewport_Left) && (dblChild_Right < dblViewport_Right);
80  bool fIsFurtherToRight = (dblChild_Right > dblViewport_Right) && (dblChild_Left > dblViewport_Left);
81  bool fIsWiderThanViewport = (dblChild_Right - dblChild_Left) > (dblViewport_Right - dblViewport_Left);
82 
83  if (!fIsFurtherToRight && !fIsFurtherToLeft)
84  //Don't change anything - the Viewport is completely visible (inside the Extent's bounds)
85  return dblViewport_Left;
86 
87  if (fIsFurtherToLeft && !fIsWiderThanViewport)
88  //The child is to be placed with its left edge equal to the left edge of the Viewport's present offset.
89  return dblChild_Left;
90 
91  //The child is to be placed with its right edge equal to the right edge of the Viewport's present offset.
92  return (dblChild_Right - (dblViewport_Right - dblViewport_Left));
93  }

303  private void UpdateOpacityMask(UIElement uieChild)
304  {
305  if (uieChild == null) return;
306 
307  //Retrieve the ScrollViewer's rectangle
308  Rect rctScrollViewerRectangle = GetScrollViewerRectangle();
309  if (rctScrollViewerRectangle == Rect.Empty) return;
310 
311  //Retrieve the child control's rectangle
312  Rect rctChildRect = GetChildRectangle(uieChild);
313 
314  if (rctScrollViewerRectangle.Contains(rctChildRect))
315  //This child is completely visible, so dump the OpacityMask.
316  uieChild.OpacityMask = null;
317  else
318  {
319  double dblPartlyVisiblePortion_OverflowToLeft = PartlyVisiblePortion_OverflowToLeft(uieChild);
320  double dblPartlyVisiblePortion_OverflowToRight = PartlyVisiblePortion_OverflowToRight(uieChild);
321 
322  if (dblPartlyVisiblePortion_OverflowToLeft < 1 && dblPartlyVisiblePortion_OverflowToRight < 1)
323  uieChild.OpacityMask = new LinearGradientBrush(
324  _gscOpacityMaskStops_TransparentOnLeftAndRight,
325  new Point(0, 0),
326  new Point(1, 0));
327  else if (dblPartlyVisiblePortion_OverflowToLeft < 1)
328  //A part of the child (to the left) remains invisible, so fade out to the left.
329  uieChild.OpacityMask = new LinearGradientBrush(
330  _gscOpacityMaskStops_TransparentOnLeft,
331  new Point(1 - dblPartlyVisiblePortion_OverflowToLeft, 0),
332  new Point(1, 0)
333  );
334  else if (dblPartlyVisiblePortion_OverflowToRight < 1)
335  //A part of the child (to the right) remains invisible, so fade out to the right.
336  uieChild.OpacityMask = new LinearGradientBrush(
337  _gscOpacityMaskStops_TransparentOnRight,
338  new Point(0, 0),
339  new Point(dblPartlyVisiblePortion_OverflowToRight, 0)
340  );
341  else
342  //This child is completely visible, so dump the OpacityMask.
343  //Actually, this part should never be reached as, in this case, the very first
344  //checkup should've resulted in the child-rect being completely contained in
345  //the SV's rect; Well, I'll leave this here anyhow (just to be save).
346  uieChild.OpacityMask = null;
347  }

509  public System.Windows.Rect MakeVisible(System.Windows.Media.Visual visual, System.Windows.Rect rectangle)
510  {
511  if (rectangle.IsEmpty || visual == null
512  || visual == this || !base.IsAncestorOf(visual))
513  { return Rect.Empty; }
514 
515  double dblOffsetX = 0;
516  UIElement uieControlToMakeVisible = null;
517  for (int i = 0; i < this.InternalChildren.Count; i++)
518  {
519  if ((Visual)this.InternalChildren[i] == visual)
520  {
521  uieControlToMakeVisible = this.InternalChildren[i];
522  dblOffsetX = getLeftEdge(this.InternalChildren[i]);
523  break;
524  }
525  }
526 
527  //Set the offset only if the desired element is not already completely visible.
528  if (uieControlToMakeVisible != null)
529  {
530  if (uieControlToMakeVisible == this.InternalChildren[0])
531  //If the first child has been selected, go to the very beginning of the scrollable area
532  dblOffsetX = 0;
533  else if (uieControlToMakeVisible == this.InternalChildren[this.InternalChildren.Count - 1])
534  //If the last child has been selected, go to the very end of the scrollable area
535  dblOffsetX = this.ExtentWidth - this.Viewport.Width;
536  else
537  dblOffsetX = CalculateNewScrollOffset(
538  this.HorizontalOffset,
539  this.HorizontalOffset + this.Viewport.Width,
540  dblOffsetX,
541  dblOffsetX + uieControlToMakeVisible.DesiredSize.Width
542  );
543 
544  SetHorizontalOffset(dblOffsetX);
545  rectangle = new Rect(this.HorizontalOffset, 0, uieControlToMakeVisible.DesiredSize.Width, this.Viewport.Height);
546  }
547 
548  return rectangle;
549  }
